# المرحلة الأولى: بناء التطبيق
FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build-env
WORKDIR /app

# 1. نسخ كل ملفات المشروع مرة واحدة لتفادي مشاكل المسارات
COPY . ./

# 2. عمل Restore باستخدام المسار المباشر لمشروع الـ API
# (سيقوم هو تلقائياً بجلب المشاريع المرتبطة به)
RUN dotnet restore "Smart_Blind_System/Smart_Blind_System.API.csproj"

# 3. بناء ونشر التطبيق في مجلد يسمى out
RUN dotnet publish "Smart_Blind_System/Smart_Blind_System.API.csproj" -c Release -o out

# المرحلة الثانية: التشغيل (Runtime)
FROM mcr.microsoft.com/dotnet/aspnet:8.0
WORKDIR /app

# نسخ الملفات التي تم بناؤها فقط من المرحلة السابقة
COPY --from=build-env /app/out .

# ضبط إعدادات الشبكة
ENV ASPNETCORE_URLS=http://+:8080
EXPOSE 8080

# نقطة الانطلاق (تأكدي من صحة اسم الـ DLL)
ENTRYPOINT ["dotnet", "Smart_Blind_System.API.dll"]
